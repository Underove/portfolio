\cvsection{2. 2025 MixUP AI Datathon : 국내 기업 철강 공정 데이터 분석}

\vspace{0.4mm}\noindent{\fontsize{9pt}{1em}\bodyfont\bfseries\color{awesome}본선 진출 및 최종 5위}\par
\projsummary{스마트팩토리 센서 데이터를 활용한 품질 불량(OK/NG) 조기 탐지 및 공정 최적화 전략 수립}
\projfig{p2.png}

\projpart{배경 및 목표}
\begin{cvitems}
  \item {\textbf{배경}\enskip 철강 제조는 고온·고압 환경에서 수많은 센서 로그가 발생하는 복잡한 공정. 미세한 변수 제어 실패가 막대한 자재 손실과 품질 저하로 직결됨.}
  \item {\textbf{목표}\enskip 위즈코어(Wizcore)의 실제 철강 공정 데이터로 양품(OK)/불량품(NG)을 판별하는 이진 분류 모델을 개발. 나아가 데이터 분석 기반의 공정 최적화·운용 전략을 제안.}
  \item {\textbf{개발 기간}\enskip 2025.05.12 -- 2025.05.18 (대면 기간 2025.05.17--18)}
  \item {\textbf{기술 스택}\enskip Python · LightGBM · Optuna · SHAP · RFECV · Scikit-learn}
\end{cvitems}

\projpart{데이터 소개}
\begin{cvitems}
  \item {\textbf{학습/평가 데이터}\enskip 제품별 센서 로그(장력·스피드 등), 생산지시 정보, 품질 결과(OK/NG).}
  \item {\textbf{메타 데이터}\enskip 공정별 설비 부하 기준(암페어 표준화), 제품 규격별 표준 운영 가이드라인.}
\end{cvitems}

\projpart{설계 과정 및 수행 내용}

\stephead{1}{물리적 인과관계 분석 및 데이터 정합성 검증}
\begin{substeps}
  \item 노이즈·이상치가 잦아, 데이터가 실제 물리 현상을 반영하는지 모델링 전 검증.
  \item '속도가 오르면 점성 저항이 커져 장력이 선형 증가'한다는 원리로 분석해, 두 변수의 높은 상관($R^2=0.7914$)을 확인.
\end{substeps}

\stephead{2}{RFECV 변수 최적화 및 도메인 피처 설계}
\begin{substeps}
  \item 27개 변수에서 RFECV로 기여도 높은 \textbf{최적 6개}(기준장력·장력1·스피드1 등)를 선별.
  \item \textbf{장력비} --- 기준 대비 편차 비율로 상대적 공정 부하를 표현.
  \item \textbf{스피드 표준편차} --- 롤러 간 속도 불균형으로 설비 동기화 오류를 포착.
  \item \textbf{측량완전결측 플래그} --- 장력·스피드 동시 0 구간의 양품률이 0\%임을 확인해, 결측이 아닌 '불량 신호'로 정의.
\end{substeps}

\stephead{3}{고정밀 품질 예측 및 데이터 불균형 최적화}
\begin{substeps}
  \item 불량(NG)이 적은 제조 현장의 불균형으로 양품 편향이 발생.
  \item Class Weight(OK:1 / NG:3)를 적용하고 비선형 패턴에 강한 LightGBM을 선정.
  \item F1-Score와 ROC-AUC를 가중 결합한 대회 전용 지표로 Optuna 튜닝을 진행.
\end{substeps}

\stephead{4}{XAI(SHAP) 기반 공정 최적화 전략 수립}
\begin{substeps}
  \item 현장 관리자가 즉시 조치할 수 있도록 판단 근거를 시각화.
  \item SHAP 분석으로 '스피드가 빠를수록 기준 장력 오차가 품질에 더 민감'한 교호작용을 도출.
  \item 이를 '고위험 구간 자동 경고'와 '실시간 피드백 제어 로직' 도입으로 비즈니스 전략화.
\end{substeps}

\stephead{5}{타겟 누수(Target Leakage) 방지}
\begin{substeps}
  \item 일자·생산지시번호 등 예측 시점에 알 수 없거나 정답과 직결된 변수 14개를 사전 제거.
  \item 실제 현장 투입 시에도 동일 성능을 유지하는 데이터 파이프라인을 구축.
\end{substeps}
