\cvsection{5. ToBig's 20th Conference : Pi-FiRi — 화재 감지 지능형 로봇 시스템}

\projsummary{라즈베리파이 로봇과 LLM Agent를 결합한 실시간 자율 순찰 및 화재 대응 시스템}
\projfigL[4.5cm]{p5.png}

\projpart{배경 및 목표}
\begin{cvitems}
  \item {\textbf{배경}\enskip 지하주차장·물류창고 등 사각지대가 많은 실내는 고정형 센서만으로 신속한 대응이 어려움. 화재 위치나 규모를 사람이 직접 판단해야 하는 위험이 존재.}
  \item {\textbf{목표}\enskip 상용 로봇 키트에 정밀 제어·환경 센서를 통합하고, LLM Agent로 화재를 실시간 탐지·분석·보고하는 '지능형 화재 감지 시스템'을 개발.}
  \item {\textbf{개발 기간}\enskip 2025.05 -- 2025.07}
  \item {\textbf{기술 스택}\enskip Python · TensorFlow · Raspberry Pi · EfficientDet-Lite1 · Flask · LangChain · Solar Pro2 · OpenCV · PID Control · Streamlit}
\end{cvitems}

\projpart{파이프라인}

\projpart{시스템 및 하드웨어 구성}
\begin{cvitems}
  \item {\textbf{EfficientDet-Lite1}\enskip 엣지 구동을 위해 TFLite 모델 중 속도(49ms)와 정확도(mAP 30.55\%)의 균형점인 'Lite1'을 선정. Roboflow Universe의 Fire\&Smoke Dataset을 수집·증강해 파인튜닝.}
  \item {\textbf{연속 감지 필터링}\enskip 단일 프레임 노이즈로 인한 오탐을 막기 위해, 0.5 임계값 이상 탐지가 1.2초 연속 유지될 때만 화재로 확정.}
  \item {\textbf{Pi-Camera}\enskip 화재·연기를 탐지하고 실시간 영상 스트림을 서버로 전송하는 로봇의 '눈' 역할.}
  \item {\textbf{GrayScale Sensor}\enskip 바닥 라인을 실시간 인식해 지정 경로를 따라가는 라인 트레이싱으로 정해진 구역을 상시 순찰.}
  \item {\textbf{DHT11 Sensor}\enskip 카메라만으로는 알 수 없는 현장 온·습도를 관제 대시보드와 실시간 Q\&A에서 함께 확인.}
  \item {\textbf{ArUco Marker (OpenCV)}\enskip 로봇의 현재 위치와 화재 발생 지점을 파악하기 위해 경로 구간별로 도입.}
\end{cvitems}

\projpart{설계 과정 및 수행 내용}

\stephead{1}{PID 제어 로직을 통한 주행 안정화 (하드웨어 한계 보완)}
\begin{substeps}
  \item PiCar-4WD 키트를 단순 if-else 조건문으로 제어하면 모터 출력 불균형으로 헌팅(Hunting)과 경로 이탈이 발생.
  \item GrayScale 센서의 라인 오차(e)를 수치화하고 PID 제어로 모터 출력을 실시간 미세 조정.
  \item 누적 오차를 반영한 I(적분) 제어로 출력 저하에 따른 주행 편향을 보정.
  \item 변화율을 감지하는 D(미분) 제어로 조향 시 떨림을 억제해 부드러운 주행을 구현.
\end{substeps}

\stephead{2}{분산 컴퓨팅 아키텍처 설계}
\begin{substeps}
  \item 엣지에서 주행 제어와 고부하 AI 추론을 동시 수행하면 연산 지연(1 FPS)으로 실시간 구동이 불가.
  \item 라즈베리파이와 로컬 PC를 같은 모바일 핫스팟(WLAN) 서브넷에 두고, Pi의 Flask API가 영상·센서 데이터를 전송하면 로컬 PC가 추론을 처리.
  \item 이 분산 구조로 약 10 FPS의 안정적 실시간성을 확보.
\end{substeps}

\stephead{3}{Agentic AI 리포팅 (Solar Pro2)}
\begin{substeps}
  \item 기존 경보는 경고음만 울려 발생 위치·규모를 즉시 파악하기 어렵고, 사람이 직접 판단해야 함.
  \item 화재 메타데이터(감지 시간·객체 크기·예측 확률·위치)를 LangChain PromptTemplate으로 동적 주입.
  \item 예측 확률과 객체 크기를 종합해 4단계 위험도[관심-주의-경계-심각] 자연어 메시지를 생성.
  \item 순찰 시작·종료 시간, 구간별 통과 로그, 화재 이벤트를 사실 기반으로 기록하는 순찰 보고서 작성(temperature=0과 다중 제약으로 환각 차단·재현성 확보).
  \item '현재 로봇 위치는?' 같은 질문에 수집 데이터 기반으로 답하는 실시간 Q\&A 제공.
\end{substeps}

\stephead{4}{Streamlit 관제 대시보드 구축}
\begin{substeps}
  \item 실시간 영상 스트리밍·센서 수치·PID 파라미터 제어·로봇 주행 상태·LLM Agent를 하나의 인터페이스에서 통합 시각화해 관제 효율을 극대화.
\end{substeps}
