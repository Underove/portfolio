\cvsection{1. 2025 DACON Bias-A-Thon : LLM 편향 대응 챌린지 (Track 2)}

\projsummary{컨텍스트 접지(Grounding)를 활용한 2단계 CoT 기반 LLM 편향 완화 및 공정성 제어}
\projfig{p1.png}

\projpart{배경 및 목표}
\begin{cvitems}
  \item {\textbf{배경}\enskip LLM은 학습 데이터에 내재된 성별·인종·정치적 편향을 그대로 학습해 차별이나 왜곡된 응답을 생성할 위험이 있음. 모델의 신뢰성을 확보하고, 실제 서비스 적용 시 발생할 수 있는 차별·왜곡·오남용 문제를 해결할 필요가 있음.}
  \item {\textbf{목표}\enskip Llama-3.1-8B 모델 환경에서 \textbf{Fine-tuning 없이} 프롬프트 엔지니어링을 활용하여, 주어진 편향 상황에 대해 중립적이고 공정한 답변을 도출.}
  \item {\textbf{개발 기간}\enskip 2025.04 -- 2025.05}
  \item {\textbf{기술 스택}\enskip Python · Llama-3.1-8B-Instruct · CoT \& Persona Prompting · Regex}
\end{cvitems}

\projpart{데이터 소개}
\begin{cvitems}
  \item {\textbf{평가 데이터}\enskip 사회적 판단 상황 본문(context), 가치 판단 질문(question), 3종 선택지(choices)로 구성.}
  \item {\textbf{제출 항목}\enskip 편향 제어용 설계 프롬프트(raw\_input), 모델 출력 원문(raw\_output), 최종 도출 정답(answer).}
\end{cvitems}

\projpart{설계 과정 및 수행 내용}
\begin{cvitems}
  \item {\textbf{\cn{1} 'AI 공정성 위원회' 페르소나 부여 및 가이드라인 구축}\enskip Llama(8B) 모델이 사전 학습된 편견(성별·인종·외모·나이 등)에 의존하지 않고 객관적 사실에만 집중하도록, 중립적인 '공정성 위원회 AI' 페르소나를 부여하고 오직 지문에 명시된 팩트(Fact)만을 근거로 삼는 컨텍스트 접지(Grounding) 원칙을 수립.}
  \item {\textbf{\cn{2} 2단계 CoT(생각--답변) 오케스트레이션 설계}\enskip 결론 도출 전 스스로 논리적 검증을 거치게 해 모델 내부의 편향된 직관을 억제. \textbf{[생각]} 단계에서 판단 근거를 먼저 작성하도록 유도하고(Few-shot으로 '지문에 명시된 근거만 인용'하는 추론 패턴 학습), \textbf{[답변]} 단계에서 선택지 중 하나를 정확히 출력하게 해 추론--결론의 정합성을 확보. 2개의 구체적 예시로 '생각'과 '답변'의 인과관계를 학습.}
  \item {\textbf{\cn{3} 구조적 출력 제어 및 정규표현식 파싱 파이프라인}\enskip 대량 데이터 처리 시의 파싱 오류를 막기 위해 Llama-3 전용 헤더·종료 토큰을 배치해 지시사항 이탈을 방지하고 출력 구조를 강제. 정규표현식(Regex) 기반 \texttt{extract\_answer} 함수로 "답변: [선택지]" 패턴을 추출·검증하고, 실패 시 마지막 줄에서 부분 매칭 $\to$ 그래도 실패하면 중립 응답('알 수 없음')을 반환하는 다단계 Fallback 로직으로 출력 방식의 일관성을 확보.}
\end{cvitems}
