% 프로젝트 공통 서식 헬퍼

%-------------------------------------------------------------------------------
% 한 줄 소개 — 본문(성과)과 같은 9pt, 강조색 세로 바
\newcommand{\projsummary}[1]{%
  \par\vspace{0.3mm}%
  \noindent{\color{awesome}\rule[-0.3mm]{2pt}{2.9mm}}\enskip
  {\fontsize{9pt}{1.25em}\bodyfontlight\upshape\color{darktext}#1}\par
  \vspace{1.6mm}%
}
% 소제목(배경·데이터·설계 등) — 사각 마커 + 10.5pt 굵은 강조색 + 전폭 얇은 구분선(그리드 정렬)
\newcommand{\projpart}[1]{%
  \par\vspace{1.3mm}%
  \noindent{\fontsize{9pt}{1em}\bodyfont\bfseries\color{awesome}#1}\par
  \vspace{0.6mm}%
}
% 설계 단계 — 노션식: [배지][굵은 제목] 헤더 줄, 그 아래 서브 불릿(줄마다 한 포인트)
\newlength{\substepsep}\setlength{\substepsep}{0.22mm}   % 불릿 사이 세로 간격
\newlength{\stepgap}\setlength{\stepgap}{0.9mm}          % 설계 단계 블록 사이 간격
\newlength{\subtwosep}\setlength{\subtwosep}{1.7mm}       % 2단계(자식) 불릿 사이 세로 간격
\newcommand{\stephead}[2]{%
  \par\vspace{\stepgap}%
  \noindent{\cn{#1}}\hspace{1.7mm}{\fontsize{9.5pt}{1em}\bodyfont\bfseries\color{darktext}#2}\par
  \vspace{-0.9mm}%
}
\newenvironment{substeps}{%
  \begingroup
  \fontsize{9pt}{1.28em}\bodyfontlight\upshape\color{text}%
  \begin{itemize}[leftmargin=6.5mm, itemsep=\substepsep, topsep=0pt, partopsep=0pt, parsep=0pt,
                  label={\textcolor{awesome}{\footnotesize\textbullet}}]%
}{%
  \end{itemize}%
  \endgroup
  \vspace{0.1mm}%
}
% 2단계 하위 목록 — 들여쓰기 축소·불릿 한 단계 작고 연하게
\newenvironment{substeps2}{%
  \vspace{1.1mm}%
  \begingroup
  \fontsize{9pt}{1.30em}\bodyfontlight\upshape\color{text}%
  \begin{itemize}[leftmargin=4.6mm, itemsep=\subtwosep, topsep=0pt, partopsep=0pt, parsep=0pt,
                  label={\textcolor{awesome!55}{\scriptsize\textbullet}}]%
}{%
  \end{itemize}%
  \endgroup
  \vspace{0.6mm}%
}
% 프로젝트 구분 표지 — "여기까지 중점 / 여기부터 그 외" 를 한 줄로 표시
\newcommand{\projdivider}[2]{%
  \par\noindent{\fontsize{12pt}{1em}\bodyfont\bfseries\color{awesome}#1}%
  \hspace{2.6mm}{\fontsize{8.5pt}{1em}\bodyfontlight\color{graytext}#2}\par
  \vspace{1.6mm}%
  {\color{awesome!30}\hrule height 0.7pt}%   % \rule 은 문단 행간이 붙어 8mm 벌어짐 → \hrule
  \vspace{-0.8mm}%                            % \cvsection 의 acvSectionTopSkip(3mm) 상쇄
}
% 성과 — 한 줄이므로 라벨 옆에 인라인(줄바꿈 없이), 본문과 같은 9pt

}
% 스크린샷 — 가운데 정렬 + 조금 작게 (높이는 선택 인자, 기본 6.2cm)
