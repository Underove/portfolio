%-------------------------------------------------------------------------------
% 프로젝트 공통 서식 헬퍼
%   \projsummary{한 줄 요약}  — 노션의 인용구(한 줄 소개)
%   \projfig{파일}            — 가운데 정렬 스크린샷 (폭 맞춤, 최대 높이 제한)
%-------------------------------------------------------------------------------
% 한 줄 소개 — 본문(성과)과 같은 9pt, 강조색 세로 바
\newcommand{\projsummary}[1]{%
  \par\vspace{0.3mm}%
  \noindent{\color{awesome}\rule[-0.3mm]{2pt}{2.9mm}}\enskip
  {\fontsize{9pt}{1.25em}\bodyfontlight\upshape\color{darktext}#1}\par
  \vspace{0.4mm}%
}
% 소제목(배경·데이터·설계 등) — 사각 마커 + 10.5pt 굵은 강조색 + 전폭 얇은 구분선(그리드 정렬)
\newcommand{\projpart}[1]{%
  \par\vspace{1.3mm}%
  \noindent{\color{awesome}\rule[0.3ex]{2.0mm}{2.0mm}}\hspace{1.5mm}%
  {\fontsize{9pt}{1em}\bodyfont\bfseries\color{awesome}#1}\par
  \vspace{0.6mm}%
}
% 설계 단계 — 노션식: [배지][굵은 제목] 헤더 줄, 그 아래 서브 불릿(줄마다 한 포인트)
\newcommand{\stephead}[2]{%
  \par\vspace{1.1mm}%
  \noindent{\cn{#1}}\hspace{1.7mm}{\fontsize{9.5pt}{1em}\bodyfont\bfseries\color{darktext}#2}\par
  \vspace{0.3mm}%
}
\newenvironment{substeps}{%
  \begingroup
  \fontsize{9pt}{1.32em}\bodyfontlight\upshape\color{text}%
  \begin{itemize}[leftmargin=6.5mm, itemsep=0.3mm, topsep=0.2mm, parsep=0pt,
                  label={\textcolor{awesome}{\footnotesize\textbullet}}]%
}{%
  \end{itemize}%
  \endgroup
  \vspace{0.1mm}%
}
% 스크린샷 — 가운데 정렬 + 조금 작게 (높이는 선택 인자, 기본 6.2cm)
\newcommand{\projfig}[2][5.8cm]{%
  \par\vspace{0.8mm}%
  {\centering\includegraphics[width=0.66\linewidth, height=#1, keepaspectratio]{#2}\par}%
  \vspace{0.3mm}%
}

%-------------------------------------------------------------------------------
% 클래스 기본값 교정
%  - cvitems 는 \cventry 밖에서 쓰면 폰트 크기를 안 잡아 문서 기본 11pt 로 커짐
%    → 성과(9pt)와 같은 9pt + 넉넉한 줄간격(1.42em)으로 고정해 크기 통일·겹침 제거
%  - 문단(성과)도 같은 9pt/1.42em
%  - 프로젝트 제목은 16pt → 13pt 로 축소
%-------------------------------------------------------------------------------
\renewenvironment{cvitems}{%
  \vspace{-1.5mm}%
  \begin{justify}%
  \fontsize{9pt}{1.42em}\bodyfontlight\upshape\color{text}%
  \begin{itemize}[leftmargin=2ex, nosep, noitemsep]%
    \setlength{\parskip}{0pt}%
    \renewcommand{\labelitemi}{\bullet}%
}{%
  \end{itemize}%
  \end{justify}%
  \vspace{1.0mm}%
}
\renewcommand*{\paragraphstyle}{\fontsize{9pt}{1.42em}\bodyfontlight\upshape\color{text}}
\renewcommand*{\sectionstyleface}[1]{{\fontsize{13pt}{1em}\bodyfont\bfseries #1}}
